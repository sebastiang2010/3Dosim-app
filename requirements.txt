# ── Launcher (PyQt5 GUI) ──────────────────────────────────────────
# Solo necesario en el Python del sistema, NO adentro de Slicer
PyQt5>=5.15,<6.0

# ── Pipeline (corre dentro de 3D Slicer) ──────────────────────────
# numpy ya incluido en Slicer 5.8.1 (no requiere instalacion)
# Pero si se necesita desde el sistema para scripts auxiliares:
numpy>=1.24

# ── AI Supervisor (OpenRouter/DeepSeek) ───────────────────────────
# Necesario adentro de Slicer para el supervisor IA
# Instalar con: <Slicer>/bin/PythonSlicer.exe -m pip install openai
openai>=1.0

# La tarea 'liver_lesions' NO soporta modo --fast (debe ser fast=False).
